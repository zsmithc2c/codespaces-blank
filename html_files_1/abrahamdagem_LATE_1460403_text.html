# Possible outputs include "The string is a palindrome.", and "The string is not a palindrome."

def pChecker(word: str):
    # Left and Right pointer
    l = 0
    r = len(word) - 1
    # Initial value until you find it's not a palindrome
    result = "The string is a palindrome."

    # Check half of the word to match first half with second
    for i in range(len(word) // 2):
        # Check for Palindrome condition and if there isn't a match set result accordingly and break
        if word[l + i].lower() != word[r-i].lower():
            result = "The string is not a palindrome."
            break
        else:
            continue
    #Return the result
    return result

#Test Cases
print(pChecker(test1))
print(pChecker(test2))
