def is_palindrome(string):
  #ignored/refined version: keeps only numbers and letters and ignores puncuation and capitalization
  ignored = ''.join(c.lower() for c in string if c.isalnum())
  # checks if ignored string is equal to its reverse
  return ignored == ignored[::-1]

# recieve user input and print result
string = input("Enter a string: ")
print("The string is a palindrome." if is_palindrome(string) else "The string is not a palindrome.")
