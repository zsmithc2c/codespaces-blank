#this function checks to see if the string is a palindrone
def s_palindrone(text):
  text = text.lower() #makes the text lowercase

  text = "".join(text.split())
  reversed_text = text[::-1] #reverses the text

  #comparing the orginal and the new text
  return text == reversed_text

#example tests to see if the code works or not
words = ["Racecar", "Hello", "lemons", "Tress", "Oceans"]

for word in words:
  if s_palindrone(word):
    print(f'"{word}" is a palindrome.')
  else:
    print(f'"{word}" is not a palindrome.')
