def is_palindrome(string):
    # Normalize the string
    cleaned_string = ""
    for char in string:
        if char.isalnum():  # Check if the character is a letter or number
            cleaned_string += char.lower()  # Convert to lowercase and add to cleaned_string

    # Check if the cleaned string is equal to its reverse
    return cleaned_string == cleaned_string[::-1]

# Get user input and display the result
user_input = input("Enter a string: ")

if is_palindrome(user_input):  # Call the function with the user's input
    print("The string is a palindrome.")
else:
    print("The string is not a palindrome.")
