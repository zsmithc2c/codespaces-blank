#Create a program that checks whether a given string is a palindrome. A palindrome is a word, phrase, number, or other sequence of characters that reads the same forwards and backwards (ignoring spaces, punctuation, and capitalization).

#Palindrome is spelled the same forwards and backwards.
#Need to compare orignal string with reversed string 
#Need to find a way to reverse a string --> Splicing
#Inputs: string
#Outputs: string (Pring a string saying if true/false)

def is_palindrome(s):
    if s.lower() == s[::-1].lower(): #Compare orignal with reversed using splicing
        print("The string is a palindrome.")
    else:
        print("The string is not a palindrome.")

is_palindrome("Racecar")
is_palindrome("Hello")
