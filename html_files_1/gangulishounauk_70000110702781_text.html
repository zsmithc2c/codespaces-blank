def isPalindrome(s):
    slower = s.lower() #Convert everything to lowercase to ignore capitalization
    removethis = " !?,.;:()&[]/-" #List of punctuation to ignore
    sready = "" #Empty string that will become the string that will be reversed after removing punctuation
    for char in slower:
        if char not in removethis:
            sready += char #Loop that iterates through the word and creates a new word from what isn't ignored
    sreverse = sready[::-1] #Reverse the final string
    if sreverse == sready: #Compares strings
        print("This string is a palindrome")
    else:
        print("This string is not a palindrome") #Prints results

isPalindrome("!R ace?.. C!aR!") #Returns true
