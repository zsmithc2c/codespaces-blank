# Ask user for a word
word = input("Enter a word: ")

# Create a new variable that reverses the word
reversed_word = ""

"""
In each iteration (using a for loop), the current character (char) 
is added to the front of reversed_word
"""
for char in word:
    reversed_word = char + reversed_word
    
if reversed_word.lower() == word.lower(): # Ensures both words are in the same format
    print("The string is a palindrome.")

elif word == "": # If word is an empty string
    print("Please type in a word.")

elif word == "?" or word == "." or word == ";" or word == ":" or word == "!": # If word has punctuation
    print("This is not a word.") 

else: # If word does not match reversed_word
    print("The string is not a palindrome.")
