#Create function to do the task of checking if a word is a palindrome
def CheckWord(word):
    #Create list to add letters into for checking 
    LetterList = []
    for letter in word:
        LetterList.append(letter)
    #Create a list for the reversed letter to compare to the normal list
    ReverseLetterList = []
    #Add each letter from the normal list in reverse to the new list
    for letter in reversed(LetterList):
        ReverseLetterList.append(letter)
    #Compare lists. If they are the same, then it is a palindrome
    if LetterList == ReverseLetterList:
        print("The string is a palindrome.")
    else:
        print("The string is not a palindrome.")

EnterWord = str(input("Enter a word here: "))
CheckWord(EnterWord)
