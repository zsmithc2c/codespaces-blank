print("I can calculate the area of a shape for you! What shape would you like me to calculate the area of?") 
shape = input("I can calculate the area of a circle, square, rectangle or triangle: ") 

if shape == "circle": 
  radius= input("What is the radius of the circle? ") 
  radius= int(radius) 
  area= 3.14 * radius * radius 
  print("The area of the circle is " + str(area)) 
elif shape == "square": 
  side= input("What is the length of the side of the square? ") 
  side= int(side) 
  area= side * side 
  print("The area of the square is " + str(area)) 
elif shape == "rectangle": 
  length= input("What is the length of the rectangle? ") 
  length= int(length) 
  width= input("What is the width of the rectangle? ") 
  width= int(width) 
  area= length * width 
  print("The area of the rectangle is " + str(area)) 
elif shape == "triangle": 
  base= input("What is the base of the triangle? ") 
  base= int(base) 
  height= input("What is the height of the triangle? ") 
  height= int(height) 
  area= base * height / 2 
  print("The area of the triangle is " + str(area)) 
else: 
  print("Sorry, I don't know how to calculate the area of that shape.")
