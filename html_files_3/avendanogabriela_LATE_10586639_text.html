def are_all_unique(string):
  #convert string to a set and compare length with original string length
  return len(set(string)) == len(string)

#ask the user for input/word
user_input = input("Enter a string to check if all characters are unique: ")

#check if the characters are unique and reply true or false
if are_all_unique(user_input):
  print("true")
else:
  print("false")
