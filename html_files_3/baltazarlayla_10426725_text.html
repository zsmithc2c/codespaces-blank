def unique_characters(text):
  s_chars = set() #to keep track of seen characters

  #looks at all characters in the string
  for char in text:
    if char in s_chars:
      return False
    s_chars.add(char)

  return True #the charaters are unique


#examples tests to see if the code works
inputs = ["abcdef", "Hello", "", "Flowers"]
for test in inputs:
  result = unique_characters(test)
  print(f'Input: "{test}" = Output: {result} ')
