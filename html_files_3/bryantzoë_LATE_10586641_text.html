'''
understand: find duplicate letters in the string

clues: count of the letter in the string would be greater than one if the letter is duplicated

assemble: 
INPUT string
FOR letter in string
    IF count of letter is > one, add to duplicates list
    IF letter is already in duplicates list, dont add again
PRINT duplicate list
'''

duplicates = []
string = input("Enter your phrase: ")
for letter in string:
    #if the letter appears more than once and is not already in the duplicate list
    if string.count(letter) > 1 and letter not in duplicates:
        #add the letter to the duplicates list
        duplicates.append(letter)
#print the duplicates
print(f'duplicates: {duplicates}')
