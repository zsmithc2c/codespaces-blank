#Create a function that determines whether all characters in a given string are unique. If the string has all unique characters, return True; otherwise, return False.

#Need to check if all characters in a string are different
#Sets --> Only contain unique characters
#Make a set using the string, and compare them
#If they have the same length, then the string is unique because no characters were removed in the set

def unique_string(s):
    string = list(s)
    check = set(string)
    if len(string) == len(check):
        return True
    else:
        return False

test = unique_string("Hello my name is Ben")    
print(f"Is the string unique? {test}")
