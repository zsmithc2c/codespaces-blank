def all_unique_characters(s):
    # Use a set to track characters we've seen
    seen_characters = set()
    for char in s:
        if char in seen_characters:
            return False
        seen_characters.add(char)
    return True

# Test cases
print(all_unique_characters("abcdef"))  # Output: True
print(all_unique_characters("hello"))   # Output: False
