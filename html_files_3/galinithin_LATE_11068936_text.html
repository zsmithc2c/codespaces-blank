def are_all_characters_unique(input_string):
    #Create an empty list to store characters 
    seen_characters = []
    
    #Loop through each character in the string
    for char in input_string:
        #Return false if the character is already in the list
        if char in seen_characters:
            return False
        #Otherwise add the character to the list
        seen_characters.append(char)
    
    #Return True if no duplicates were found
    return True
