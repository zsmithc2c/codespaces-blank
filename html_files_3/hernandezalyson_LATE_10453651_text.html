def is_unique(string):
    
    # Removes any spaces and lowercases the string
    new = string.replace(" ", "").lower()
    
    # Main part of this line is the "set(string)"
    #   set(string) removes and duplicate characters, and then it's length is set equal to the length of the original string
    return len(set(new)) == len(new)
    
#Case 1
print(is_unique("abcdef")) 

#Case 2
print(is_unique("hello")) 

#Case 3
print(is_unique("Sub dermatoglyphic"))
