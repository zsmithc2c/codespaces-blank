from time import time

def is_unique(string):
    # trim whitespace
    string = string.replace(" ", "")
    # seen list to track seen characters
    seen = []

    # loop through each character
    for char in string:
        # quit and return false the moment we see a repeat
        # check to see if the current character is in our seen list
        if char in seen:
            return False
        seen.append(char)
    # if we reach this we can assume all chars are unique and return true
    return True

# less lines
def is_unique_alt(string):
    clean_str = string.replace(" ", "")
    if len(set(clean_str)) != len(clean_str):
        return False
    return True

start = time()
is_unique("abcdefghijlkmonpdlgjk;lkgjl       ;aszjdfal;sjdg;l          akjsdg;lakgjsd;a")
print(time() - start)

start = time()
is_unique_alt("abcdefghijlkmonpdlgjk;lkgjl       ;aszjdfal;sjdg;l          akjsdg;lakgjsd;a")
print(time() - start)

# test values
