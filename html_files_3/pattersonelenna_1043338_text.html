string = input('Enter a string: ')
#make empty list
check = []
result = True
for letter in string:
    if letter in check:
        result = False
        break
    else:
        result = True
    #add to list end of loop so that it runs through each letter through the sequence
    check.append(letter)

if result == False:
    print("False")
else:
    print("True")
