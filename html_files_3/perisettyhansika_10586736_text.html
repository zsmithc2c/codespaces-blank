public class Unique {
    private String word;

    // Constructor to initialize word
    public Unique(String w) {
        word = w;
    }

    // Method to check if the string has unique characters
    public boolean isUnique() {
        for (int i = 0; i < word.length() - 1; i++) {
            // Compare current character with the next character
            if (word.substring(i, i + 1).equals(word.substring(i + 1, i + 2))) {
                return false; // Return false if a duplicate is found
            }
        }
        return true; // Return true if no duplicates are found after the loop
    }
    
    public static void main(String[] args) {
        Unique string = new Unique("hello");
        System.out.println(string.isUnique()); // Output: false (because 'l' repeats)

        Unique string2 = new Unique("abc");
        System.out.println(string2.isUnique()); // Output: true (no duplicates)
    }

}
