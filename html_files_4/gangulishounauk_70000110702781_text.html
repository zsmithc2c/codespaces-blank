import random
randNumber = random.randint(1, 100) #create random number
def guessingGame():
    try:
        guess = int(input("Guess a number"))
    except: #handles errors in input
        print("Please try again")
        guessingGame()
    if guess == randNumber:
        print("Congratulations!") #end of the game
        exit()
    elif guess < randNumber: #recurses if the guess is too low
        print("Too low! Try again!") 
        guessingGame()
    elif guess > randNumber: #recurses if the guess is too high
        print("Too high! Try again!")
        guessingGame()
guessingGame() #begins the program
