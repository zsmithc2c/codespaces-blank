# Pseudocode
# 1. Generate a random number between 1 and 100.
# 2. Use a while loop so person can keep guessing until its right.
# 3. Compare the guess to the random number.

import random
def number_guessing_game():
    # Generate a random number
    target_number = random.randint(1, 100)
    print("I have chosen a number between 1 and 100. Guess what it is!")

    while True:
        try:
            user_guess = int(input("Enter your guess: "))

            # Compare the guess to the target number
            if user_guess == target_number:
                print("You guessed the number correctly!")
                break
              #tell them if too high or low
            elif user_guess > target_number:
                print("Too high!")
            else:
                print("Too low!")
