#import random to use random library

import random
user_num = int(input('Enter a number 1-100: '))
comp_num = random.randint(1, 100)

#loop to make sure that it runs through until correct number

while user_num < 1 or user_num > 100:
    user_num = int(input('Please enter a number 1-100: '))

#loop until user == computer

while user_num != comp_num:
    if user_num > comp_num:
        user_num = int(input('Your number is too high, guess again: '))
    elif user_num < comp_num:
        user_num = int(input('Your number is too low, guess again: '))

print('You got the correct number!')
