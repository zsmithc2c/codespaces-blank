import random # we use the random library for the game 

number = random.randint(1, 100)#It gives use a random int number between 1 to 100 for the guessing game
print("Guess a number between 1 and 100!")

# Use a forever loop to guess the correct number and break from the loop until we get it right

while True:
    guess = int(input("Your guess: "))
    if guess < number:
        print("Too low!")
    elif guess > number:
        print("Too high!")
    else:
        print("You got it!")
        break
