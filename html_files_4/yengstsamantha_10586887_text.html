def good_number():

        num = input("Please guess a number between 1 and 100: ")

            num = int(num)
        except: # error
            print("Something went wrong. Please try again.")
            print()
        else: # no error
            if (num < 1 or num > 100):
                print("Please chose a number between 1 and 100.")
                print()
            else:
                return num
